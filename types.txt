% ps object types

/Ring: <<
    /components         [ Token ]
    /circumference      float
    /radius             float
    /full_radius        float
    /rtype              name         % options: array dict xarray
>>

/Token: <<
    /name name
    /otype      name
    /xflag      bool
    /aflag      bool
    /position   float
    /angle      float
    /parent     Ring

    (if aflag)
    /offset     float
    /ring_angle float
    /ring       Ring
>>
